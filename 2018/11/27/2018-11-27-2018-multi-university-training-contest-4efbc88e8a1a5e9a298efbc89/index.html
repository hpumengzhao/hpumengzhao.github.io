Problem B. Harvest of Apples

题意：计算c(n,0)+c(n,1)+c(n,2)+…c(n,m).
solve:令s(n,m)=c(n,0)+c(n,1)+c(n,2)+…c(n,m).则s(n,m)=2s(n-1,m)+c(n-1,m).
莫队离线查询即可。

[code lang=cpp]

#include<bits/stdc++.h>

using namespace std;
typedef long long ll;
const ll maxn = 1e5+100;
const ll mod = 1e9+7;
ll fact[maxn],inv[maxn],factinv[maxn];
ll anss[maxn];
struct query{
ll l,r,id;
}e[maxn];
void init(){
fact[0]=inv[1]=factinv[0]=inv[0]=fact[1]=factinv[1]=1;
for(int i=2;i<=maxn-10;i++){
fact[i]=(fact[i-1]%modi%mod)%mod;
inv[i]=((mod-mod/i)%modinv[mod%i]%mod)%mod;
factinv[i]=(factinv[i-1]%modinv[i]%mod)%mod;
}
}
ll c(ll n,ll m){
return (fact[n]%modfactinv[m]%modfactinv[n-m]%mod)%mod;
}
ll bl,ans;
ll n,m;
bool cmp1(query a,query b){
return (a.l/bl==b.l/bl)?a.r<b.r:a.l<b.l;
}
void add1(int pos){
ans=((2%modans%mod)%mod-c(pos-1,m)+mod)%mod;
}
void remove1(int pos){
ans=((ans%mod+c(pos-1,m))%mod*inv[2])%mod;
}
void add2(int pos){
ans=(ans%mod+c(n,pos))%mod;;
}
void remove2(int pos){
ans=(ans%mod-c(n,pos)+mod)%mod;
}
int main(){

ll t;
n=1;m=0;
ll L,R;
init();
//    cout<<inv[100000]<<endl;
scanf("%lld",&t);
bl=sqrt(t);
for(int i=1;i<=t;i++){
     scanf("%lld %lld",&e[i].l,&e[i].r);
     e[i].id=i;
}
sort(e+1,e+t+1,cmp1);
ans=1;
for(int i=1;i<=t;i++){
    L=e[i].l;R=e[i].r;
    while(n<L) add1(++n);
    while(n>L) remove1(n--);
    while(m<R) add2(++m);
    while(m>R) remove2(m--);
    anss[e[i].id]=ans;
}
for(int i=1;i<=t;i++){
    printf("%lld\n",anss[i]);
}
return 0;

}

Problem L. Graph Theory Homework

题意：一个n个顶点的完全图，i点的点权为wi，两点之间的权值为sqrt(abs(wi-wj))，求1-n的最短路。
solve：数学推导一下就会发现最短路=sqrt(abs(w1-wn));
